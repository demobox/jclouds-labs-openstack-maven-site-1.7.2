57  /**
58   * Provides access to the Rackspace Cloud Files CDN API features.
59   *
60   * <h3>NOTE</h3>
61   * Before a container can be CDN enabled, it must exist in the storage system. 
62   * To CDN enable the container, perform PUT request against it using the <code>publicURL</code> 
63   * noted in the service catalog for Cloud Files during Authentication and set the 
64   * <code>X-CDN-Enabled</code> header to <code>true</code>.
65   * 
66   * @author Jeremy Daggett
67   * 
68   * @see {@link CloudFilesApi#cdnApiInRegion(String)}
69   */
70  @RequestFilters(AuthenticateRequest.class)
71  @Consumes(APPLICATION_JSON)
72  public interface CDNApi extends Closeable {
73  
74     /**
75      * Lists up to 10,000 CDN containers.
76      * 
77      * @return a list of CDN enabled containers ordered by name.
78      */
79     @Named("cdn:list")
80     @GET
81     @QueryParams(keys = {"format", "enabled_only"}, values = {"json", "true"})
82     @Fallback(EmptyFluentIterableOnNotFoundOr404.class)
83     @Path("/")
84     FluentIterable<CDNContainer> list();
85     
86     /**
87      * Lists CDN containers, with the given options.
88      * 
89      * @param options
90      *           the options to control output.
91      * 
92      * @return a list of CDN enabled containers ordered by name.
93      */
94     @Named("cdn:list")
95     @GET
96     @QueryParams(keys = {"format", "enabled_only"}, values = {"json", "true"})
97     @Fallback(EmptyFluentIterableOnNotFoundOr404.class)
98     @Path("/")
99     FluentIterable<CDNContainer> list(ListContainerOptions options);
100 
101    /**
102     * Gets the specified CDN Container.
103     * 
104     * @param containerName
105     *           the name of the CDN Container
106     * 
107     * @return the CDNContainer or null, if not found.
108     */
109    @Named("cdn:get")
110    @HEAD
111    @ResponseParser(ParseCDNContainerFromHeaders.class)
112    @Fallback(NullOnNotFoundOr404.class)
113    @Path("/{container}")
114    @Nullable
115    CDNContainer get(@PathParam("container") String containerName);
116 
117    /**
118     * Enables the {@link CDNContainer}.
119     * 
120     * @param containerName
121     *           corresponds to {@link CDNContainer#getName()}.
122     *           
123     * @return the CDN container {@link URI} or {@code null}, if not found.
124     */
125    @Named("cdn:enable")
126    @PUT
127    @ResponseParser(ParseCDNContainerURIFromHeaders.class)
128    @Fallback(NullOnNotFoundOr404.class)
129    @Path("/{containerName}")
130    @Headers(keys = CDN_ENABLED, values = "true")
131    @Nullable
132    URI enable(@PathParam("containerName") String containerName);
133 
134    /**
135     * Enables the {@link CDNContainer} with a TTL.
136     * 
137     * @param containerName
138     *           corresponds to {@link CDNContainer#name()}.
139     * @param ttl
140     *           the TTL for the CDN Container.
141     *           
142     * @return the CDN container {@link URI} or {@code null}, if not found.
143     */
144    @Named("cdn:enable")
145    @PUT
146    @ResponseParser(ParseCDNContainerURIFromHeaders.class)
147    @Fallback(NullOnNotFoundOr404.class)
148    @Path("/{containerName}")
149    @Headers(keys = CDN_ENABLED, values = "true")
150    @Nullable
151    URI enable(@PathParam("containerName") String containerName, 
152          @HeaderParam(CDN_TTL) int ttl);
153 
154    /**
155     * Disables the {@link CDNContainer}.
156     * 
157     * @param containerName
158     *           corresponds to {@link CDNContainer#name()}.
159     *           
160     * @return {@code true} if the container was disabled, {@code false} if not.
161     */
162    @Named("cdn:disable")
163    @PUT
164    @Fallback(FalseOnNotFoundOr404.class)
165    @Path("/{containerName}")
166    @Headers(keys = CDN_ENABLED, values = "False")
167    boolean disable(@PathParam("containerName") String containerName);
168 
169    /**
170     * Purges an object from the CDN.
171     * 
172     * @param containerName
173     *           corresponds to {@link CDNContainer#name()}.
174     * @param objectName
175     *           the object in the {@link CDNContainer} to purge.
176     * @param emails
177     *           the email addresses to notify after purging.
178     * 
179     * @return {@code true} if the object was successfully purged, {@code false} if not.
180     */
181    @Named("cdn:purge")
182    @DELETE
183    @Fallback(FalseOnNotFoundOr404.class)
184    @Path("/{containerName}/{objectName}")
185    @Headers(keys = CloudFilesHeaders.CDN_PURGE_OBJECT_EMAIL, values = "{email}")
186    boolean purgeObject(@PathParam("containerName") String containerName, 
187          @PathParam("objectName") String objectName, 
188          @BinderParam(BindCDNPurgeEmailAddressesToHeaders.class) Iterable<String> emails);
189 
190    /**
191     * Updates a CDN container with the supplied {@link UpdateCDNContainerOptions} options.
192     * 
193     * @param containerName
194     *           corresponds to {@link CDNContainer#name()}.
195     *           
196     * @param options
197     *           the {@link UpdateCDNContainerOptions} options.
198     */
199    @Named("cdn:update")
200    @POST
201    @Fallback(FalseOnNotFoundOr404.class)
202    @Path("/{containerName}")
203    boolean update(@PathParam("containerName") String containerName, UpdateCDNContainerOptions options);
