52     public void testGetAndListRouters() {
53        for (String zone : api.getConfiguredZones()) {
54           Set<? extends ReferenceWithName> references = api.getRouterExtensionForZone(zone).get().list().concat().toSet();
55           Set<? extends Router> routers = api.getRouterExtensionForZone(zone).get().listInDetail().concat().toSet();
56  
57           assertNotNull(references);
58           assertNotNull(routers);
59           assertEquals(references.size(), routers.size());
60  
61           for (Router router : routers) {
62              assertNotNull(router.getName());
63              assertTrue(references.contains(ReferenceWithName.builder().id(router.getId()).tenantId(router.getTenantId()).name(router.getName()).build()));
64  
65              Router retrievedRouter = api.getRouterExtensionForZone(zone).get().get(router.getId());
66              assertEquals(router, retrievedRouter);
67           }
68        }

71     public void testCreateUpdateAndDeleteRouter() {
72        for (String zone : api.getConfiguredZones()) {
73           RouterApi routerApi = api.getRouterExtensionForZone(zone).get();
74           NetworkApi networkApi = api.getNetworkApiForZone(zone);
75           SubnetApi subnetApi = api.getSubnetApiForZone(zone);
76  
77           Network network = networkApi.create(CreateNetworkOptions.builder().name("jclouds-network-test").external(true).networkType(NetworkType.LOCAL).build());
78           assertNotNull(network);
79  
80           Subnet subnet = subnetApi.create(network.getId(), 4, "192.168.0.0/16");
81           assertNotNull(subnet);
82  
83           Router ref = routerApi.create(CreateRouterOptions.builder().name("jclouds-router-test")
84              .externalGatewayInfo(ExternalGatewayInfo.builder().networkId(network.getId()).build()).build());
85           assertNotNull(ref);
86  
87           Router router = routerApi.get(ref.getId());
88  
89           assertEquals(router.getId(), ref.getId());
90           assertEquals(router.getName(), "jclouds-router-test");
91           assertEquals(router.getExternalGatewayInfo().getNetworkId(), network.getId());
92           assertTrue(routerApi.update(router.getId(), UpdateRouterOptions.builder().name("jclouds-router-test-rename").build()));
93  
94           router = routerApi.get(ref.getId());
95  
96           assertEquals(router.getId(), ref.getId());
97           assertEquals(router.getName(), "jclouds-router-test-rename");
98  
99           ReferenceWithName ref2 = routerApi.create(CreateRouterOptions.builder().name("jclouds-router-test2")
100             .externalGatewayInfo(ExternalGatewayInfo.builder().networkId(network.getId()).build()).build());
101          assertNotNull(ref2);
102 
103          assertTrue(routerApi.delete(ref.getId()));
104          assertTrue(routerApi.delete(ref2.getId()));
105          assertTrue(subnetApi.delete(subnet.getId()));
106          assertTrue(networkApi.delete(network.getId()));
107       }
