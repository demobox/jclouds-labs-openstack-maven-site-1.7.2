353    @Override
354    @AfterClass(groups = { "integration", "live" })
355    public void tearDown() {
356       for (String zone : api.getConfiguredZones()) {
357          GroupApi groupApi = api.getGroupApiForZone(zone);
358          for (Group group : created.get(zone)) {
359             PolicyApi policyApi = api.getPolicyApiForZoneAndGroup(zone, group.getId());
360             if(policyApi == null)continue;
361             for(ScalingPolicy sgr : policyApi.list()) {
362                if(!policyApi.delete(sgr.getId())) {
363                   System.out.println("Could not delete an autoscale policy after tests!");
364                }
365             }
366 
367             List<CreateScalingPolicy> scalingPolicies = Lists.newArrayList();
368 
369             CreateScalingPolicy scalingPolicy = CreateScalingPolicy.builder()
370                   .cooldown(3)
371                   .type(ScalingPolicyType.WEBHOOK)
372                   .name("0 machines")
373                   .targetType(ScalingPolicyTargetType.DESIRED_CAPACITY)
374                   .target("0")
375                   .build();
376             scalingPolicies.add(scalingPolicy);
377 
378             FluentIterable<ScalingPolicy> scalingPolicyResponse = policyApi.create(scalingPolicies);
379             String policyId = scalingPolicyResponse.first().get().getId();
380 
381             Uninterruptibles.sleepUninterruptibly(10, TimeUnit.SECONDS);
382 
383             policyApi.execute(policyId);
384 
385             Uninterruptibles.sleepUninterruptibly(10, TimeUnit.SECONDS);
386 
387             if (!groupApi.delete(group.getId()))
388                System.out.println("Could not delete an autoscale group after tests!");
389          }
390       }
391       super.tearDown();
