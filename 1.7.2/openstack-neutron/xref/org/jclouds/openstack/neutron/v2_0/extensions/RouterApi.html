55  /**
56   * Provides synchronous access to Router operations on the OpenStack Neutron API.
57   * <p/>
58   * A logical entity for forwarding packets across internal subnets and NATting them on external
59   * networks through an appropriate external gateway.
60   *
61   * @author Nick Livens
62   * @see <a href=
63   *      "http://docs.openstack.org/api/openstack-network/2.0/content/router_ext.html">api doc</a>
64   */
65  @Path("/v2.0/routers")
66  @RequestFilters(AuthenticateRequest.class)
67  @Consumes(MediaType.APPLICATION_JSON)
68  public interface RouterApi {
69  
70     /**
71      * Returns the list of all routers currently defined in Neutron for the current tenant. The list provides the unique
72      * identifier of each router configured for the tenant
73      *
74      * @return the list of all router references configured for the tenant.
75      */
76     @Named("router:list")
77     @GET
78     @ResponseParser(ParseRouters.class)
79     @Transform(ParseRouters.ToPagedIterable.class)
80     @Fallback(Fallbacks.EmptyPagedIterableOnNotFoundOr404.class)
81     @QueryParams(keys = {"fields", "fields", "fields"}, values = {"id", "tenant_id", "name"})
82     PagedIterable<? extends ReferenceWithName> list();
83  
84     /**
85      * @see <a href="http://docs.openstack.org/api/openstack-network/2.0/content/pagination.html">api doc</a>
86      */
87     @Named("router:list")
88     @GET
89     @ResponseParser(ParseRouters.class)
90     @Fallback(EmptyPaginatedCollectionOnNotFoundOr404.class)
91     @QueryParams(keys = {"fields", "fields", "fields"}, values = {"id", "tenant_id", "name"})
92     PagedIterable<? extends ReferenceWithName> list(PaginationOptions options);
93  
94     /**
95      * Returns all routers currently defined in Neutron for the current tenant.
96      *
97      * @return the list of all routers configured for the tenant
98      */
99     @Named("router:list")
100    @GET
101    @ResponseParser(ParseRouterDetails.class)
102    @Transform(ParseRouterDetails.ToPagedIterable.class)
103    @Fallback(Fallbacks.EmptyPagedIterableOnNotFoundOr404.class)
104    PagedIterable<? extends Router> listInDetail();
105 
106    /**
107     * @see <a href="http://docs.openstack.org/api/openstack-network/2.0/content/pagination.html">api doc</a>
108     */
109    @Named("router:list")
110    @GET
111    @ResponseParser(ParseRouterDetails.class)
112    @Fallback(EmptyPaginatedCollectionOnNotFoundOr404.class)
113    PagedIterable<? extends Router> listInDetail(PaginationOptions options);
114 
115    /**
116     * Returns the specific router.
117     *
118     * @param id the id of the router to return
119     * @return Router or null if not found
120     */
121    @Named("router:get")
122    @GET
123    @Path("/{id}")
124    @SelectJson("router")
125    @Fallback(Fallbacks.NullOnNotFoundOr404.class)
126    @Nullable
127    Router get(@PathParam("id") String id);
128 
129    /**
130     * Create a new router
131     *
132     * @param options optional arguments
133     * @return the newly created router
134     */
135    @Named("router:create")
136    @POST
137    @SelectJson("router")
138    @MapBinder(CreateRouterOptions.class)
139    Router create(CreateRouterOptions... options);
140 
141    /**
142     * Update a router
143     *
144     * @param id the id of the router to update
145     * @param options the attributes to update
146     * @return true if update successful, false if not
147     */
148    @Named("router:update")
149    @PUT
150    @Path("/{id}")
151    @MapBinder(UpdateRouterOptions.class)
152    @Fallback(Fallbacks.FalseOnNotFoundOr404.class)
153    boolean update(@PathParam("id") String id, UpdateRouterOptions... options);
154 
155    /**
156     * Deletes the specified router
157     *
158     * @param id the id of the router to delete
159     * @return true if delete successful, false if not
160     */
161    @Named("router:delete")
162    @DELETE
163    @Path("/{id}")
164    @Fallback(Fallbacks.FalseOnNotFoundOr404.class)
165    boolean delete(@PathParam("id") String id);
166 
167    /**
168     * Add a interface to a router to connect to the specified subnet
169     *
170     * @param routerId the id of the router to create the interface at
171     * @param subnetId the id of the subnet to connect with the interface
172     * @return the newly-created router interface
173     */
174    @Named("router:addInterfaceForSubnet")
175    @PUT
176    @Path("/{id}/add_router_interface")
177    @MapBinder(EmptyOptions.class)
178    RouterInterface addInterfaceForSubnet(@PathParam("id") String routerId, @PayloadParam("subnet_id") String subnetId);
179 
180    /**
181     * Add a interface to a router to connect to the specified port
182     *
183     * @param routerId the id of the router to create the interface at
184     * @param portId the id of the port to connect with the interface
185     * @return the newly-created router interface
186     */
187    @Named("router:addInterfaceForPort")
188    @PUT
189    @Path("{id}/add_router_interface")
190    @MapBinder(EmptyOptions.class)
191    RouterInterface addInterfaceForPort(@PathParam("id") String routerId, @PayloadParam("port_id") String portId);
192 
193    /**
194     * Remove the interface where the specified subnet is connected to
195     *
196     * @param routerId the id of the router to remove the interface from
197     * @param subnetId the id of the subnet to disconnect from the interface
198     */
199    @Named("router:removeInterfaceForSubnet")
200    @PUT
201    @Path("/{id}/remove_router_interface")
202    @MapBinder(EmptyOptions.class)
203    boolean removeInterfaceForSubnet(@PathParam("id") String routerId, @PayloadParam("subnet_id") String subnetId);
204 
205    /**
206     * Remove the interface where the specified port is connected to
207     *
208     * @param routerId the id of the router to remove the interface from
209     * @param portId the id of the port to disconnect from the interface
210     */
211    @Named("router:removeInterfaceForPort")
212    @PUT
213    @Path("{id}/remove_router_interface")
214    @MapBinder(EmptyOptions.class)
215    boolean removeInterfaceForPort(@PathParam("id") String routerId, @PayloadParam("port_id") String portId);
