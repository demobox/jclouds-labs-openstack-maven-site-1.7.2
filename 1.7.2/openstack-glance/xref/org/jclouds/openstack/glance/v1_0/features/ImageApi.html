58  /**
59   * Image Services
60   * 
61   * @author Adrian Cole
62   * @author Adam Lowe
63   * @see <a href="http://glance.openstack.org/glanceapi.html">api doc</a>

68  @RequestFilters(AuthenticateRequest.class)
69  public interface ImageApi {
70  
71     /**
72      * List all images (IDs, names, links)
73      * 
74      * @return all images (IDs, names, links)
75      */
76     @GET
77     @Consumes(APPLICATION_JSON)
78     @Path("/images")
79     @RequestFilters(AuthenticateRequest.class)
80     @ResponseParser(ParseImages.class)
81     @Transform(ParseImages.ToPagedIterable.class)
82     @Fallback(EmptyPagedIterableOnNotFoundOr404.class)
83     PagedIterable<? extends Image> list();
84  
85     @GET
86     @Consumes(APPLICATION_JSON)
87     @Path("/images")
88     @RequestFilters(AuthenticateRequest.class)
89     @ResponseParser(ParseImages.class)
90     @Fallback(EmptyPaginatedCollectionOnNotFoundOr404.class)
91     PaginatedCollection<? extends Image> list(ListImageOptions options);
92  
93     /**
94      * List all images (all details)
95      * 
96      * @return all images (all details)
97      */
98     @GET
99     @Consumes(APPLICATION_JSON)
100    @Path("/images/detail")
101    @RequestFilters(AuthenticateRequest.class)
102    @ResponseParser(ParseImageDetails.class)
103    @Transform(ParseImageDetails.ToPagedIterable.class)
104    @Fallback(EmptyPagedIterableOnNotFoundOr404.class)
105    PagedIterable<? extends ImageDetails> listInDetail();
106 
107    @GET
108    @Consumes(APPLICATION_JSON)
109    @Path("/images/detail")
110    @RequestFilters(AuthenticateRequest.class)
111    @ResponseParser(ParseImageDetails.class)
112    @Fallback(EmptyPaginatedCollectionOnNotFoundOr404.class)
113    PaginatedCollection<? extends ImageDetails> listInDetail(ListImageOptions options);
114 
115    /**
116     * Return metadata about an image with id
117     */
118    @HEAD
119    @Path("/images/{id}")
120    @ResponseParser(ParseImageDetailsFromHeaders.class)
121    @Fallback(NullOnNotFoundOr404.class)
122    @Nullable
123    ImageDetails get(@PathParam("id") String id);
124 
125    /**
126     * Return image data for image with id
127     */
128    @GET
129    @Path("/images/{id}")
130    @Fallback(NullOnNotFoundOr404.class)
131    @Nullable
132    InputStream getAsStream(@PathParam("id") String id);
133 
134    /**
135     * Create a new image
136     * 
137     * @return detailed metadata about the newly stored image
138     */
139    @POST
140    @Path("/images")
141    @Produces(APPLICATION_OCTET_STREAM)
142    @SelectJson("image")
143    @Consumes(APPLICATION_JSON)
144    ImageDetails create(@HeaderParam("x-image-meta-name") String name, Payload payload, CreateImageOptions... options);
145 
146    /**
147     * Reserve a new image to be uploaded later
148     * 
149     * @return detailed metadata about the newly stored image
150     * @see #upload
151     */
152    @POST
153    @Path("/images")
154    @SelectJson("image")
155    @Consumes(APPLICATION_JSON)
156    ImageDetails reserve(@HeaderParam("x-image-meta-name") String name, CreateImageOptions... options);
157 
158    /**
159     * Upload image data for a previously-reserved image
160     * <p/>
161     * If an image was previously reserved, and thus is in the queued state, then
162     * image data can be added using this method. If the image already as data
163     * associated with it (e.g. not in the queued state), then you will receive a
164     * 409 Conflict exception.
165     * 
166     * @param imageData
167     *           the new image to upload
168     * @param options
169     *           can be used to adjust the metadata stored for the image in the
170     *           same call
171     * @return detailed metadata about the updated image
172     * @see #reserve
173     */
174    @PUT
175    @Path("/images/{id}")
176    @Produces(APPLICATION_OCTET_STREAM)
177    @SelectJson("image")
178    @Consumes(APPLICATION_JSON)
179    ImageDetails upload(@PathParam("id") String id, Payload imageData, UpdateImageOptions... options);
180 
181    /**
182     * Adjust the metadata stored for an existing image
183     * 
184     * @return detailed metadata about the updated image
185     */
186    @PUT
187    @Path("/images/{id}")
188    @SelectJson("image")
189    @Consumes(APPLICATION_JSON)
190    ImageDetails update(@PathParam("id") String id, UpdateImageOptions... options);
191 
192    /**
193     * Delete the image with the specified id
194     * 
195     * @return true if successful
196     */
197    @DELETE
198    @Path("/images/{id}")
199    @Fallback(FalseOnNotFoundOr404.class)
200    boolean delete(@PathParam("id") String id);
